// 1. GitHub上で分割管理するファイルの一覧 const dataFiles = [ 'yougo-a.html', // あ行 'yougo-ka.html', // か行 'yougo-sa.html', // さ行 'yougo-ta.html', // た行 'yougo-na.html', // な行 'yougo-ha.html', // は行 'yougo-ma.html', // ま行 'yougo-ya.html', // や行 'yougo-ra.html', // ら行 'yougo-wa.html', // わ行 'yougo-num.html' // 英数 ]; let allTerms = []; // すべての用語データを格納する配列 let currentRow = 'すべて'; // 現在選択されている50音フィルター（初期値） // 2. ページ開いた瞬間にすべての行のHTMLファイルを裏でロードする async function loadAllData() { for (const file of dataFiles) { try { const response = await fetch(file); if (!response.ok) continue; const text = await response.text(); const parser = new DOMParser(); const doc = parser.parseFromString(text, 'text/html'); // 各ファイルから用語の塊を引っ張る const items = doc.querySelectorAll('.term-data'); items.forEach(item => { allTerms.push({ title: item.dataset.title, yomi: item.dataset.yomi, row: item.dataset.row, // ★「あ」「か」などの行の識別用 desc: item.innerHTML.trim() // 説明文（HTMLタグも含めて保持） }); }); } catch (e) { console.error("読み込みエラー:", file); } } // データが溜まったら、初期表示として「すべて」を表示 renderTerms(); } // 3. 画面に用語を描画する処理（検索とフィルターを同時にかける） function renderTerms() { const query = document.getElementById('search-input').value.toLowerCase().trim(); const resultsContainer = document.getElementById('results'); // 表示先のエリア resultsContainer.innerHTML = ''; // 検索文字 と 50音ボタン の両方で絞り込み const filtered = allTerms.filter(item => { const matchesSearch = item.title.toLowerCase().includes(query) || item.yomi.includes(query) || item.desc.toLowerCase().includes(query); const matchesRow = (currentRow === 'すべて') || (item.row === currentRow); return matchesSearch && matchesRow; }); // 該当件数の表示を更新（画像右側にある「該当件数: ○件」のパーツ） document.getElementById('match-count').innerText = `${filtered.length}件`; // 画面に出力（現在の素敵なデザインのHTML構造に合わせて出力してください） filtered.forEach(item => { const div = document.createElement('div'); div.className = 'term-card'; // お使いのカードのクラス名 div.innerHTML = `

${item.title}

${item.yomi}
${item.desc}
`; resultsContainer.appendChild(div); }); } // 4. 50音ボタンが押されたときのイベント処理（例） // ※実際のボタンのHTMLに合わせて調整してください document.querySelectorAll('.filter-btn').forEach(btn => { btn.addEventListener('click', (e) => { currentRow = e.target.textContent; // 「あ」「か」などを取得 // アクティブなボタンの色を変える処理（現在のスタイルを適用） document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active')); e.target.classList.add('active'); renderTerms(); // 再描画 }); }); // 入力があったら検索を実行 document.getElementById('search-input').addEventListener('input', renderTerms); // 読み込み開始 window.addEventListener('DOMContentLoaded', loadAllData);
