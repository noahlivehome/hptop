// 800個の完全ユニークなオブジェクトを格納する配列 const finalGlossaryData = []; // 50単語 × 16軸 ＝ 800個を完全に生成 matrixData.core_terms.forEach(baseItem => { matrixData.extensions.forEach(ext => { // 用語名から（）をカットして綺麗に結合 const cleanTerm = baseItem.term.split('（')[0]; const finalTerm = cleanTerm + ext.suffix; const finalKana = baseItem.kana + ext.kana; const finalDesc = `${baseItem.term}${ext.desc}`; finalGlossaryData.push({ "term": finalTerm, "kana": finalKana, "group": baseItem.group, "category": baseItem.category, "desc": finalDesc }); }); }); // 読みがなの50音順に美しくソート finalGlossaryData.sort((a, b) => a.kana.localeCompare(b.kana, 'ja')); // これで「finalGlossaryData」にジャスト800個の有効なJSON配列が展開されます console.log(finalGlossaryData.length); // 800 console.log(JSON.stringify(finalGlossaryData, null, 2));
